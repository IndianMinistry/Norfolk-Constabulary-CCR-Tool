<script>

  // ── Roleplay name (persisted in localStorage) ──
  let roleplayName = localStorage.getItem("rpName") || "UNKNOWN";

  function openRP() {
    document.getElementById("rpNameInput").value = roleplayName;
    document.getElementById("roleplayModal").style.display = "flex";
  }

  function saveRPName() {
    const val = document.getElementById("rpNameInput").value.trim();
    roleplayName = val || "UNKNOWN";
    localStorage.setItem("rpName", roleplayName);
    document.getElementById("roleplayModal").style.display = "none";
  }

  // Close modal on overlay click
  document.getElementById("roleplayModal").addEventListener("click", function(e) {
    if (e.target === this) this.style.display = "none";
  });

  // ── Tab switching ──
  function openTab(id, btn) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.getElementById(id).classList.add('active');
    btn.classList.add('active');
  }

  // ── Bubble toggles ──
  function toggleBubble(el) {
    el.classList.toggle("active");
  }

  function getBubbleValues(id) {
    return [...document.querySelectorAll(`[data-id="${id}"] .bubble.active`)]
      .map(x => x.innerText).join(", ");
  }

  // ── Conditional show/hide ──
  function toggleCaller() {
    document.getElementById("callerOther").style.display =
      document.getElementById("callerType").value === "Other" ? "block" : "none";
  }

  function togglePCVehicle() {
    document.getElementById("pcVehicleOther").style.display =
      document.getElementById("pcVehicle").value === "Other" ? "block" : "none";
  }

  // ── 999 ──
  function generate999() {
    let caller = document.getElementById("callerType").value;
    if (caller === "Other") caller = document.getElementById("callerOther").value.trim() || "Other";

    const details  = document.getElementById("callDetails").value.trim();
    const location = document.getElementById("callLocation").value.trim();

    const text =
`999 CALL REPORT

Call Details: ${details || "(none)"}

Location: ${location || "(none)"}

Caller: ${caller}
Threat Level: ${document.getElementById("threatLevel").value}${document.getElementById("callRemarks").value.trim() ? `\n\nRemarks: ${document.getElementById("callRemarks").value.trim()}` : ""}

CAD generated by ${roleplayName}`;

    document.getElementById("output999").innerText = text;
    copyText(text);
  }

  // ── FTS ──
  function generateFTS() {
    const pcVehicleEl = document.getElementById("pcVehicle");
    const pcVehicleVal = pcVehicleEl.value === "Other"
      ? document.getElementById("pcVehicleOther").value.trim() || "Other"
      : pcVehicleEl.value;

    const quals = getBubbleValues("pcQual");

    const text =
`FAILURE TO STOP REPORT

Reporting Officer: ${document.getElementById("reportingOfficer").value.trim() || "(none)"}

Vehicle: ${document.getElementById("ftsVehicle").value.trim() || "(none)"}

Originating Location: ${document.getElementById("ftsLocation").value.trim() || "(none)"}

Vehicle Density: ${document.getElementById("vehicleDensity").value}
Pedestrian Density: ${document.getElementById("pedDensity").value}
Visibility: ${document.getElementById("visibility").value}
Road Conditions: ${document.getElementById("roadConditions").value}

DRA Level: ${document.getElementById("draLevel").value}
Estimated Speeds: ${document.getElementById("estimatedSpeeds").value.trim() || "(none)"}

Qualifications: ${quals || "None selected"}

Pursuit Commander Vehicle: ${pcVehicleVal}${document.getElementById("ftsRemarks").value.trim() ? `\n\nRemarks: ${document.getElementById("ftsRemarks").value.trim()}` : ""}

CAD generated by ${roleplayName}`;

    document.getElementById("outputFTS").innerText = text;
    copyText(text);
  }

  // ── FIREARMS ──
  function generateFirearms() {
    const sections = [];

    function addSection(title, id) {
      const val = getBubbleValues(id);
      if (val) sections.push(`${title}: ${val}`);
    }

    addSection("Building Tactics",    "bt");
    addSection("Vehicle Tactics",     "vt");
    addSection("Pedestrian Tactics",  "pt");
    addSection("Deployment",          "dt");
    addSection("Declaration Criteria","dc");

    const text =
`DECLARED FIREARMS INCIDENT

Incident Details: ${document.getElementById("incidentDetails").value.trim() || "(none)"}

Location: ${document.getElementById("incidentLocation").value.trim() || "(none)"}

Assigned OFC: ${document.getElementById("assignedOFC").value.trim() || "(none)"}

${sections.length ? sections.join("\n\n") : "(No tactics or criteria selected)"}${document.getElementById("firearmsRemarks").value.trim() ? `\n\nRemarks: ${document.getElementById("firearmsRemarks").value.trim()}` : ""}

This is being declared as a Firearms Incident by ${roleplayName}, the current on-duty Tactical Firearms Commander located within the Brisley Police Headquarters' Control Room.`;

    document.getElementById("outputFirearms").innerText = text;
    copyText(text);
  }

  // ── OBSERVATIONS ──
  function generateObs() {
    const text =
`OBSERVATIONS REPORT

Reporting Officer: ${document.getElementById("obsOfficer").value.trim() || "(none)"}
Type: ${document.getElementById("obsType").value.trim() || "(none)"}
Visible Gender: ${document.getElementById("obsGender").value}
IC Code: ${document.getElementById("obsIC").value}

Last Seen: ${document.getElementById("obsLastSeen").value.trim() || "(none)"}
Markers: ${document.getElementById("obsMarkers").value.trim() || "(none)"}

Remarks: ${document.getElementById("obsRemarks").value.trim() || "(none)"}

CAD generated by ${roleplayName}`;

    document.getElementById("outputObs").innerText = text;
    copyText(text);
  }

  // ── RESETS ──
  function reset999() {
    document.querySelectorAll("#call999 input, #call999 textarea").forEach(x => x.value = "");
    document.querySelectorAll("#call999 select").forEach(x => x.selectedIndex = 0);
    document.getElementById("callerOther").style.display = "none";
    document.getElementById("output999").innerText = "";
  }

  function resetFTS() {
    document.querySelectorAll("#fts input, #fts textarea").forEach(x => x.value = "");
    document.querySelectorAll("#fts select").forEach(x => x.selectedIndex = 0);
    document.querySelectorAll("#fts .bubble").forEach(b => b.classList.remove("active"));
    document.getElementById("pcVehicleOther").style.display = "none";
    document.getElementById("outputFTS").innerText = "";
  }

  function resetFirearms() {
    document.querySelectorAll("#firearms input, #firearms textarea").forEach(x => x.value = "");
    document.querySelectorAll("#firearms .bubble").forEach(b => b.classList.remove("active"));
    document.getElementById("outputFirearms").innerText = "";
  }

  function resetObs() {
    document.querySelectorAll("#obs input, #obs textarea").forEach(x => x.value = "");
    document.querySelectorAll("#obs select").forEach(x => x.selectedIndex = 0);
    document.getElementById("outputObs").innerText = "";
  }

  // ── CLIPBOARD ──
  async function copyText(text) {
    try {
      await navigator.clipboard.writeText(text);
    } catch (err) {
      // Fallback for environments without clipboard API
      const ta = document.createElement("textarea");
      ta.value = text;
      ta.style.position = "fixed";
      ta.style.opacity = "0";
      document.body.appendChild(ta);
      ta.select();
      document.execCommand("copy");
      document.body.removeChild(ta);
    }
    showToast();
  }

  function showToast() {
    const toast = document.getElementById("copyToast");
    toast.style.display = "block";
    clearTimeout(toast._timer);
    toast._timer = setTimeout(() => { toast.style.display = "none"; }, 2000);
  }

</script>
